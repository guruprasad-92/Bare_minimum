00327  * To initialize the device, and call the main() routine.

00329 void Reset_Handler(void)
00330 {
00331     uint32_t *pSrc, *pDest;
00332 
00333     /* Initialize the relocate segment */
00334     pSrc = &_etext;
00335     pDest = &_srelocate;
00336 
00337     if (pSrc != pDest) {
00338         for (; pDest < &_erelocate;)
00339             *pDest++ = *pSrc++;
00340     }
00341 
00342     /* Clear the zero segment */
00343     for (pDest = &_szero; pDest < &_ezero;)
00344         *pDest++ = 0;
00345 
00346     /* Set the vector table base address */
00347     pSrc = (uint32_t *) & _sfixed;
00348     SCB->VTOR = ((uint32_t) pSrc & SCB_VTOR_TBLOFF_Msk);
00349 
00350 #ifdef ENABLE_TCM
00351 #ifndef FFT_DEMO
00352     // 32 Kb
00353     EFC->EEFC_FCR = (EEFC_FCR_FKEY_PASSWD | EEFC_FCR_FCMD_CGPB
00354                      | EEFC_FCR_FARG(8));
00355 #else
00356     // 128 Kb
00357     EFC->EEFC_FCR = (EEFC_FCR_FKEY_PASSWD | EEFC_FCR_FCMD_SGPB

00364 #else
00365     EFC->EEFC_FCR = (EEFC_FCR_FKEY_PASSWD | EEFC_FCR_FCMD_CGPB
00366                      | EEFC_FCR_FARG(8));
